			clientes[index][3] = saldo

			index++
			PARE

			DECLARE int qtdeUsuarios
			LEIA qtdeUsuario

				DECLARE int idade
				DECLARE float saldo
				LEIA nome, idade
				REPITA

					SE saldo < 0 FAÇA
						MOSTRE "Saldo inválido"
				ENQUANTO saldo < 0
				id++
				clientes[index][0] = id
